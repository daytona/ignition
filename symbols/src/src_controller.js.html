1 /**
  2  * Base Controller class. This is used to create all Controllers for
  3  * {@link Ignition} Objects. If you want, you can extend the base Controller using
  4  * <code>jQuery.extend()</code>.
  5  *
  6  * @example $i.m.MyController = new Ignition.Controller({
  7  *   index: function() {
  8  *     // Load some data with MyModel, and present it using MyView:
  9  *     $i.m.MyModel.load($i.v.MyView.index);
 10  *   }
 11  * });
 12  *
 13  * @example // Calling:
 14  * $i.m.MyView.index();
 15  *
 16  * @example // Extend the base Controller:
 17  * jQuery.extend(Ignition.Controller.prototype, {
 18  *   myCustomMethod: function() {
 19  *     alert('This method will be available for all Controllers.');
 20  *   }
 21  * });
 22  *
 23  * @constructor
 24  * @param {Object} [methods] A collection of methods to extend the base Controller with
 25  */
 26 Ignition.Controller = function(name, methods) {
 27   this.name = name;
 28   $.extend(this, methods);
 29 };
