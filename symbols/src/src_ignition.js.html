1 /**
  2  * The base class constructor.
  3  * You can specify Modules to load for the resulting object.
  4  *
  5  * @example var $i = new Ignition({ modules: ['UrlManager', 'HistoryDispatcher'] });
  6  *
  7  * @class The main class is intended to be instantiated as a 'wrapper' object,
  8  * which you would then use to work with your models, views and controllers.
  9  * <br/><br/>
 10  * You can use Modules (the built-in
 11  * <code>{@link Ignition.Modules.UrlManager}</code> and
 12  * <code>{@link Ignition.Modules.HistoryDispatcher}</code>) to get some, for these
 13  * kinds of XHR applications, commonly needed functionality for your
 14  * Ignition-based app.
 15  * <br/><br/>
 16  * Usage information can be found at
 17  * <a href="http://wiki.github.com/daytona/ignition">http://wiki.github.com/daytona/ignition</a>
 18  *
 19  * @constructor
 20  * @param [options] A object of options
 21  * @param {Array} options.modules An array of modules to load for the resulting Object
 22  */
 23 var Ignition = window.Ignition = function(options) {
 24   var options = $.extend({
 25     modules: []
 26   }, options);
 27 
 28   for (var i = 0; i < options.modules.length; i++) {
 29     var module = options.modules[i];
 30     if (Ignition.Modules[module] && (typeof Ignition.Modules[module].addTo == 'function')) Ignition.Modules[module].addTo(this);
 31   }
 32 };
 33 
 34 /**
 35  * A map of the available Modules to load for an Ignition Object
 36  * @property
 37  */
 38 Ignition.Modules = {};
 39 
 40 Ignition.prototype = {
 41   /**
 42    * A map of the available Models
 43    * @property
 44    */
 45   m: {},
 46 
 47   /**
 48    * A map of the available Views
 49    * @property
 50    */
 51   v: {},
 52 
 53   /**
 54    * A map of the available Controllers
 55    * @property
 56    */
 57   c: {},
 58 
 59   /**
 60    * Add a new Model. An optional collection of methods
 61    * will be used to extend the base Model with <code>jQuery.extend()</code>.
 62    *
 63    * @example $i.m('MyModel', {
 64    *   load: function(callback) {
 65    *     // Load some content and peform the callback method
 66    *   },
 67    *   beforeSend: function() {
 68    *     // Overwriting the base Model's default beforeSend method
 69    *   }
 70    * });
 71    *
 72    * @example // Calling:
 73    * $i.m.MyModel.load();
 74    *
 75    * @param {String} name The name of the new Model
 76    * @param {Object} [methods] A collection of methods to extend the base Model with
 77    *
 78    * @returns {Ignition.Model}
 79    */
 80   m: function(name, methods) {
 81     this.m[name] = new Ignition.Model(methods);
 82     return this.m[name];
 83   },
 84 
 85   /**
 86    * Add a new View. An optional collection of methods
 87    * will be used to extend the base View with <code>jQuery.extend()</code>.
 88    *
 89    * @example $i.v('MyView', {
 90    *   index: function(data) {
 91    *     // Generate DOM elements based on data
 92    *   }
 93    * });
 94    *
 95    * @example // Calling:
 96    * $i.v.MyView.index(data);
 97    *
 98    * @param {String} name The name of the new View
 99    * @param {Object} [methods] A collection of methods to extend the base View with
100    *
101    * @returns {Ignition.View}
102    */
103   v: function(name, methods) {
104     this.v[name] = new Ignition.View(methods);
105     return this.v[name];
106   },
107 
108   /**
109    * Add a new Controller. An optional collection of
110    * methods will be used to extend the base Controller with <code>jQuery.extend()</code>.
111    *
112    * @example $i.c('MyController', {
113    *   index: function() {
114    *     // Load some data with MyModel, and present it using MyView:
115    *     $i.m.MyModel.load($i.v.MyView.index);
116    *   }
117    * });
118    *
119    * @example // Calling:
120    * $i.c.MyController.index();
121    *
122    * @param {String} name The name of the new Controller
123    * @param {Object} [methods] A collection of methods to extend the base Controller with
124    *
125    * @returns {Ignition.Controller}
126    */
127   c: function(name, methods) {
128     this.c[name] = new Ignition.Controller(name, methods);
129     return this.c[name];
130   }
131 };
