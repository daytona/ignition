1 /**
  2  * Base View class. This is used to create all Views for
  3  * {@link Ignition} Objects. If you want, you can extend the base View using
  4  * <code>jQuery.extend()</code>.
  5  *
  6  * @example $i.m.MyView = new Ignition.View({
  7  *   index: function(data) {
  8  *     // // Generate DOM elements based on data
  9  *   }
 10  * });
 11  *
 12  * @example // Calling:
 13  * $i.m.MyView.index();
 14  *
 15  * @example // Extend the base View:
 16  * jQuery.extend(Ignition.View.prototype, {
 17  *   myCustomMethod: function() {
 18  *     alert('This method will be available for all Views.');
 19  *   }
 20  * });
 21  *
 22  * @constructor
 23  * @param {Object} [methods] A collection of methods to extend the base View with
 24  */
 25 Ignition.View = function(methods) {
 26   $.extend(this, methods);
 27 };
