1 /**
  2  * Create DOM elements and start timers. An iframe is created if the user is
  3  * using IE, to support detection of back/forward navigation. Timers to check
  4  * if the window hash (or the iframe contents) have changed, and if so perform
  5  * the specified actions, are set using <code>setInterval()</code>.
  6  *
  7  * @class This class makes it possible to watch
  8  * the <code>window.location.hash</code> for changes, performing custom
  9  * actions when specific patterns are matched. This could potentially be used,
 10  * without modifications, independently of the Ignition framework aswell.
 11  *
 12  * @constructor
 13  * @param {Object} [options]
 14  * @param {Number} [options.interval=200] How often to check the location hash
 15  */
 16 Ignition.Modules.HistoryDispatcher = function(options) {
 17   var options = $.extend({
 18     interval: 200
 19   }, options);
 20 
 21   var self = this;
 22 
 23   this._window.bind('hashchange', function() { self._triggerAction(); });
 24 
 25   this._engage(options);
 26 };
 27 
 28 /**
 29  * Clean a <code>window.location.hash</code> string. (Remove the '#' prefix.)
 30  * @param {String} hash A hash string (most probably from <code>window.location.hash</code>)
 31  */
 32 Ignition.Modules.HistoryDispatcher.cleanHash = function(hash) {
 33   return hash.replace(/^#/, '');
 34 };
 35 
 36 /**
 37  * Adds a HistoryDispatcher to the specified ({@link Ignition}) object.
 38  * The Dispatcher is assigned to the <code>dispatcher</code> property of the
 39  * object, with aliases to
 40  * <code>{@link Ignition.Modules.HistoryDispatcher#addRoute}</code> as
 41  * <code>object.addRoutes</code> and
 42  * <code>{@link Ignition.Modules.HistoryDispatcher#go}</code> as
 43  * <code>object.go</code>.
 44  *
 45  * @param {Ignition} object A previously initialised <code>Ignition</code> object.
 46  */
 47 Ignition.Modules.HistoryDispatcher.addTo = function(object) {
 48   /** @ignore */
 49   object.dispatcher = new Ignition.Modules.HistoryDispatcher();
 50   /** @ignore */
 51   object.addRoute   = function() { return this.dispatcher.addRoute.apply(this.dispatcher, arguments); };
 52   /** @ignore */
 53   object.go         = function() { return this.dispatcher.go.apply(this.dispatcher, arguments); };
 54 };
 55 
 56 Ignition.Modules.HistoryDispatcher.prototype = {
 57   /** @property */
 58   _window: $(window),
 59 
 60   /**
 61    * A map of the current routes that have been added to the object.
 62    * @property
 63    */
 64   routes: {},
 65 
 66   /** @property */
 67   _hash: '',
 68 
 69   /**
 70    * An Interval that checks if the location Hash has changed.
 71    * @property
 72    */
 73   timer: null,
 74 
 75   /** @property */
 76   __ie: false,
 77 
 78   /**
 79    * A reference to the DOM element of the generated iframe (if on IE).
 80    * @property
 81    */
 82   iframe: null,
 83 
 84   /**
 85    * A timer that watches the iframe for changes (if on IE).
 86    * @property
 87    */
 88   iframetimer: null,
 89 
 90   /** @function */
 91   _cleanHash: Ignition.Modules.HistoryDispatcher.cleanHash,
 92 
 93   /**
 94    * Add a new route. Specifies a pattern to be matched, and an action
 95    * (function) to be called when it is. The pattern can be a RegExp string,
 96    * and will be automatically surrounded by <code>^</code> and <code>$</code>
 97    * notations. RegExp back references will be used as arguments on the
 98    * specified function.
 99    *
100    * @example // Add a new route to a previously defined HistoryDispatcher object:
101    * dispatcher.addRoute('posts/(\\d+)', function(id) {
102    *   alert('Matched post with ID '+id);
103    * });
104    *
105    * @param {String} pattern A string, which can be a RegExp string. If so,
106    *     you need to 'double escape', e.g. <code>"\d"</code> should be written
107    *     as <code>"\\d"</code>.
108    * @param {Function} action The action to be called when the pattern
109    *     is matched.
110    */
111   addRoute: function(pattern, action) {
112     var action = action || function(){};
113     this.routes[pattern] = action;
114   },
115 
116   /**
117    * Go to a hash of your choice. The same as clicking a link with
118    * <code>href="#(hash)"</code>.
119    *
120    * @example dispatcher.go('posts/123');
121    * 
122    * @param {String} hash
123    */
124   go: function(hash) {
125     window.location.hash = '#'+hash;
126   },
127 
128   /** @function */
129   _triggerAction: function() {
130     var hash = this._cleanHash(window.location.hash);
131     this._hash = hash;
132 
133     this._updateIFrame(hash);
134 
135     for (pattern in this.routes) {
136       var re = new RegExp('^'+pattern+'$');
137       var matches = re.exec(hash);
138       if (matches) {
139         this.routes[pattern].apply(pattern, matches.slice(1,matches.length));
140         return;
141       }
142     }
143   },
144 
145   /** @function */
146   _checkIFrameState: function() {
147     if (!this.iframe) {
148       var that = this;
149       setTimeout(function() {
150         that._checkIFrameState();
151       }, 10);
152       return;
153     }
154 
155     var doc = this.iframe.contentWindow.document;
156     var state = this._hash;
157     var elem = doc.getElementById('hash');
158     var newState = elem ? elem.innerText : '';
159 
160     if (newState != state) {
161       this.iframetext = newState;
162       window.location.hash = newState;
163       this._window.trigger('hashchange');
164     }
165   },
166 
167   /** @function */
168   _updateIFrame: function(hash) {
169     if (!this.__ie) return;
170 
171     var content = '<html><body><div id="hash">'+hash+'</div></body></html>';
172 
173     if (!this.iframe) {
174       var that = this;
175       setTimeout(function() {
176         that._updateIFrame(hash);
177       }, 10);
178       return content;
179     }
180 
181 
182     var doc = this.iframe.contentWindow.document;
183     var elem = doc.getElementById('hash');
184     var currentState = elem ? elem.innerText : '';
185     if (hash != currentState) {
186       doc.open();
187       doc.write(content);
188       doc.close();
189     }
190 
191     return content;
192   },
193 
194   /** @function */
195   _engage: function(options) {
196     var self = this;
197 
198     this.timer = setInterval(function() {
199       var hash = self._cleanHash(window.location.hash);
200       if (hash != self._hash) self._window.trigger('hashchange');
201     }, options.interval);
202 
203     if (this.__ie = ($.browser.msie && parseInt($.browser.version) < 8)) {
204       $(function() {
205         var iframe = $('<iframe id="Ignition_history_iframe" style="display: none;"></iframe>');
206         $('body').prepend(iframe);
207         self.iframe = iframe.get(0);
208         self.iframestate = self._hash || '';
209       });
210 
211       this.iframetimer = setInterval(function() {
212         self._checkIFrameState();
213       }, options.interval);
214     }
215   },
216 
217   /**
218    * Stop watching for history changes. Stops all timer. Use this if you don't
219    * want to watch for history changes any more, for example if a page that
220    * doesn't use the Ignition system is loaded.
221    */
222   disengage: function() {
223     for (timer in [this.timer, this.iframetimer]) {
224       if (timer) {
225         clearInterval(timer);
226         timer = null;
227       }
228     }
229   }
230 };
