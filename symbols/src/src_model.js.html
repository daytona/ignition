1 /**
  2  * Base Model class. This is used to create all Models for
  3  * {@link Ignition} Objects. If you want, you can extend the base Model using
  4  * <code>jQuery.extend()</code>.
  5  *
  6  * @example $i.m.MyModel = new Ignition.Model({
  7  *   load: function(callback) {
  8  *     // Load some content and peform the callback method
  9  *   },
 10  *   beforeSend: function() {
 11  *     // Overwriting the base Model's default beforeSend method
 12  *   }
 13  * });
 14  *
 15  * @example // Calling:
 16  * $i.m.MyModel.load();
 17  *
 18  * @example // Extend the base Model:
 19  * jQuery.extend(Ignition.Model.prototype, {
 20  *   myCustomMethod: function() {
 21  *     alert('This method will be available for all Models.');
 22  *   }
 23  * });
 24  *
 25  * @constructor
 26  * @param {Object} [methods] A collection of methods to extend the base Model with
 27  */
 28 Ignition.Model = function(methods) {
 29   if (methods && !(typeof methods == 'object')) {
 30     Ignition.error('Model creation error');
 31   } else {
 32     $.extend(this, methods);
 33   }
 34 };
 35 
 36 Ignition.Model.prototype = {
 37   /**
 38    * Get data from the specified URL as JSON data.
 39    * This is a wrapper for jQuery.ajax(), with a pre-set dataType of 'json'.
 40    * This method will use the callbacks you specify in the options, if any.
 41    * If you haven't specified any such callbacks for this method, the default
 42    * callbacks for the current Model object will be used, if available.
 43    *
 44    * Any additional options will be passed to the jQuery.ajax() function.
 45    *
 46    * $i.m.MyModel('/items.json', {
 47    *   success: function() {
 48    *     // Perform some actions when the XHR call is successful
 49    *   }
 50    * });
 51    *
 52    * @see http://docs.jquery.com/Ajax/jQuery.ajax
 53    *
 54    * @param {Object} [options]
 55    * @param {Function} options.success Called when the XHR call is successful
 56    * @param {Function} options.error Called when the XHR call isn't successful
 57    * @param {Function} options.beforeSend Called before starting the XHR transaction
 58    * @param {Function} options.complete Called after the XHR call is done, regardless if it is successful or not
 59    */
 60   json: function(url, options) {
 61     var model = this;
 62     var options = $.extend({
 63       success: model.success,
 64       error: model.error,
 65       beforeSend: model.beforeSend,
 66       complete: model.complete
 67     }, options, {
 68       url: url,
 69       dataType: 'json'
 70     });
 71 
 72     return $.ajax(options);
 73   },
 74 
 75   /**
 76    * Default success callback for Ajax calls. Overwrite this with your own.
 77    * @see jQuery API documentation
 78   */
 79   success: function() {},
 80 
 81   /**
 82    * Default error callback for Ajax calls. Overwrite this with your own.
 83    * @see jQuery API documentation
 84   */
 85   error: function() {},
 86 
 87   /**
 88    * Default beforeSend callback for Ajax calls. Overwrite this with your own.
 89    * @see jQuery API documentation
 90   */
 91   beforeSend: function() {},
 92 
 93   /**
 94    * Default complete callback for Ajax calls. Overwrite this with your own.
 95    * @see jQuery API documentation
 96   */
 97   complete: function() {}
 98 };
