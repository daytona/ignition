1 /**
  2  * @class This class makes it easy to handle URLs (like JSON URLs in
  3  * a complex XHR application), including placeholders.
  4  *
  5  * @constructor
  6  */
  7 Ignition.Modules.UrlManager = function() {
  8 };
  9 
 10 /**
 11  * Adds a UrlManager to the specified ({@link Ignition}) object.
 12  * The manager is assigned to the <code>urlManager</code> property of the
 13  * object, with aliases to <code>{@link Ignition.Modules.UrlManager#add}</code>
 14  * as <code>object.addUrl</code> and
 15  * <code>{@link Ignition.Modules.UrlManager#compile}</code> as
 16  * <code>object.getUrl</code>.
 17  *
 18  * @param {Ignition} object A previously initialised <code>Ignition</code> object.
 19  */
 20 Ignition.Modules.UrlManager.addTo = function(obj) {
 21   obj.urlManager = new Ignition.Modules.UrlManager();
 22   /** @ignore */
 23   obj.addUrl = function() { return this.urlManager.add.apply(obj.urlManager, arguments); };
 24   /** @ignore */
 25   obj.getUrl = function() { return this.urlManager.compile.apply(obj.urlManager, arguments); };
 26 };
 27 
 28 Ignition.Modules.UrlManager.prototype = {
 29   /** @property */
 30   _urls: {},
 31 
 32   /**
 33    * Add a new URL to the handler. The name is used to reference the URL later
 34    * on when using <code>compile</code>, the path is the actual path to the
 35    * resource (including placeholders with the syntax
 36    * <code>:placeholder</code>). Default values for the placeholder(s) can be
 37    * specified aswell.
 38    *
 39    * @example manager.add('post', '/posts/:category/:id.json', { category: 'news', id: 1 });
 40    *
 41    * @param {String} name Reference name for the URL
 42    * @param {String} path The path to the resource
 43    * @param {Object} [placeholders] Default values for the placeholders
 44    *     specified in <code>path</code>
 45    */
 46   add: function(name, path, defaults) {
 47     this._urls[name] = { path: path, defaults: defaults || {} }
 48   },
 49 
 50   /**
 51    * Retrieve the compiled version of a URL. If the the URL was originally
 52    * added with placeholders, you can specify replacements for these when
 53    * retrieving it.
 54    *
 55    * @example // Retrieve the JSON URL for a post with from the category 'tutorials' with ID 123
 56    * manager.compile('post', { category: 'tutorials', id: 123 });
 57    * // => "/posts/tutorials/123.json"
 58    *
 59    * @param {String} name Reference name for a previously added URL
 60    * @param {Object} [replacements] Replacements for placeholders in the URL
 61    */
 62   compile: function(name, replacements) {
 63     if (r = this._urls[name]) {
 64       var replacements = $.extend({}, r.defaults, replacements);
 65       var result = ''+r.path;
 66       for (key in replacements) {
 67         var re = new RegExp(':'+key, "g");
 68         result = result.replace(re, replacements[key]);
 69       }
 70       return result;
 71     } else {
 72       throw("Error: Url not found!");
 73     }
 74   }
 75 };
